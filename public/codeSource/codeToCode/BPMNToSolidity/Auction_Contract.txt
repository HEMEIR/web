// SPDX-License-Identifier: MIT
pragma solidity 0.8.20;

contract VotingContract {
    address public immutable owner;
    mapping(address => bool) public voters;
    uint256[] public candidateList;
    bool public votingLive;

    event VoterRegistered(address voter);
    event VoteCasted(address voter, uint256 candidate);
    event VotingStarted();
    event VotingEnded();

    constructor(uint256[] memory candidateList_) {
        owner = msg.sender;
        candidateList = candidateList_;
        votingLive = false;
    }

    modifier onlyOwner() {
        require(msg.sender == owner, "Only owner can call this function.");
        _;
    }

    modifier onlyRegisteredVoter() {
        require(voters[msg.sender], "Only registered voters can call this function.");
        _;
    }

    function registerVoter(address voter) public onlyOwner {
        require(!voters[voter], "Voter is already registered.");
        voters[voter] = true;
        emit VoterRegistered(voter);
    }

    function startVoting() public onlyOwner {
        require(!votingLive, "Voting has already started.");
        votingLive = true;
        emit VotingStarted();
    }

    function endVoting() public onlyOwner {
        require(votingLive, "Voting is not live.");
        votingLive = false;
        emit VotingEnded();
    }

    function vote(uint256 candidateNum) public onlyRegisteredVoter {
        require(votingLive, "Voting is not live.");
        require(candidateNum < candidateList.length, "Invalid candidate number.");
        candidateList[candidateNum]++;
        emit VoteCasted(msg.sender, candidateNum);
    }

    function totalVotesFor(uint256 candidateNum) public view returns (uint256) {
        require(candidateNum < candidateList.length, "Invalid candidate number.");
        return candidateList[candidateNum];
    }
}

// SPDX-License-Identifier: MIT
pragma solidity ^0.8.20;  // 更新到一个稳定且无严重已知问题的版本

contract MeatSalesContract {
    address payable public immutable buyer;
    address payable public immutable seller;
    uint256 public immutable effDate;
    uint256 public immutable qnt;
    uint256 public immutable qlt;
    uint256 public immutable delDueDateDays;
    uint256 public immutable amt;
    uint256 public immutable payDueDate;
    uint256 public immutable intRate;

    bool public payment = false;
    bool public latePay = false;
    bool public delivery = false;
    bool public contractEnded = false;

    event OrderDelivered(uint256 amount);
    event PaymentMade(uint256 amount);
    event LatePayment(uint256 interestAmount);
    event ContractTerminated(uint256 interestAmount);

    modifier onlySeller() {
        require(msg.sender == seller, "Only seller can call this function");
        _;
    }

    modifier onlyBuyer() {
        require(msg.sender == buyer, "Only buyer can call this function");
        _;
    }

    constructor(
        address payable _buyer,
        address payable _seller,
        uint256 _effDate,
        uint256 _qnt,
        uint256 _qlt,
        uint256 _delDueDateDays,
        uint256 _amt,
        uint256 _payDueDate,
        uint256 _intRate
    ) {
        require(_buyer != address(0), "Buyer address cannot be zero");
        require(_seller != address(0), "Seller address cannot be zero");
        buyer = _buyer;
        seller = _seller;
        effDate = _effDate;
        qnt = _qnt;
        qlt = _qlt;
        delDueDateDays = _delDueDateDays;
        amt = _amt;
        payDueDate = _payDueDate;
        intRate = _intRate;
    }

    function endContract() external onlySeller {
        require(block.timestamp > payDueDate && !payment && latePay, "Invalid conditions for contract termination");
        require(!contractEnded, "Contract already ended");
        contractEnded = true; // 更新状态，防止重入

        uint256 interestAmount = (amt * intRate) / 100;
        emit ContractTerminated(interestAmount); // 先触发事件
        seller.transfer(interestAmount); // 后进行外部调用
    }

    function deliverGoods() external onlySeller {
        require(block.timestamp < delDueDateDays && !delivery && payment, "Invalid conditions for goods delivery");
        delivery = true;
        emit OrderDelivered(amt);
    }

    function payPrice() external payable onlyBuyer {
        require(block.timestamp < payDueDate && !payment, "Invalid conditions for payment");
        require(msg.value == amt, "Incorrect value for payment");
        payment = true;
        emit PaymentMade(amt);
    }

    function payInitRate() external payable onlyBuyer {
        require(block.timestamp > payDueDate && !payment && !latePay, "Invalid conditions for interest payment");
        uint256 interestAmount = (amt * intRate) / 100;
        require(msg.value == interestAmount, "Incorrect value for interest payment");
        latePay = true;
        emit LatePayment(interestAmount);
    }
}
